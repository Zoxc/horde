1//! An API for quiescent state based reclamation.
2//!
3//! Threads that read shared lock-free data call [pin] to mark the duration of the critical
4//! section. Destruction of removed data is postponed with [defer_unchecked] and later driven by
5//! [collect], which advances the global quiescent-state cycle.
6//!
7//! Long-lived threads that stop interacting with lock-free structures should call [release] so
8//! they no longer delay reclamation.
9

31/// Monotonic event counter used to notify threads that reclamation state changed.
32static EVENTS: AtomicUsize = AtomicUsize::new(0);

34/// Represents a proof that no deferred callbacks will run for the lifetime `'a`.
35///
36/// A `Pin` is handed to the closure passed to [pin] and can be used to access data structures in
37/// a lock-free manner while the current thread remains pinned.
38#[derive(Clone, Copy)]
39pub struct Pin<'a> {
40    _private: PhantomData<&'a ()>,
41}
42
43/// Schedules a closure to run after all threads leave their current pinned regions.
44///
45/// The closure will be called by the [collect] method.
46///
47/// This deferred callback must not call [pin] or [collect].
48///
49/// Deferred callbacks run in a later collection cycle on whichever thread performs that collection.
50/// They are typically used to destroy or free data that was removed from a lock-free structure.
51pub fn defer(f: impl FnOnce() + Send + 'static) {
52    COLLECTOR.lock().defer(Box::new(f));
53}

55/// Schedules a closure to run after all threads leave their current pinned regions.
56///
57/// The closure will be called by the [collect] method.
58///
59/// This deferred callback must not call [pin] or [collect].
60///
61/// Deferred callbacks run in a later collection cycle on whichever thread performs that collection.
62/// They are typically used to destroy or free data that was removed from a lock-free structure.
63///
64/// # Safety
65/// This method is unsafe since the closure is not required to be `'static`.
66/// It's up to the caller to ensure the closure does not access freed memory.
67/// A `move` closure is recommended to avoid accidental references to stack variables.
68pub unsafe fn defer_unchecked<F>(f: F)
69where
70    F: FnOnce(),
71    F: Send,
72{
73    unsafe {
74        let f: Box<dyn FnOnce() + Send> = Box::new(f);
75        let f: Box<dyn FnOnce() + Send + 'static> = mem::transmute(f);
76
77        COLLECTOR.lock().defer(f);
78    }
79}

82    static DATA: Data = const {
83        Data {
84            state: Cell::new(State::Unregistered),
85            seen_events: Cell::new(0),
86        }
87    };
88    static EXIT_GUARD: ExitGuard = const { ExitGuard };

91#[derive(Clone, Copy, PartialEq, Eq, Debug)]
92enum State {
93    // Valid states to immediately pin in.
94    // This is placed on top so `pin` can check these states with a single comparison.
95    /// Registered with the collector and allowed to enter a pinned region immediately.
96    Registered,
97    /// Currently inside a pinned region.
98    Pinned,
99
100    // Not valid states to immediately pin in
101    /// Not currently known to the collector.
102    Unregistered,
103    /// Running deferred callbacks as part of a collection pass.
104    Collecting,
105}
106
107/// Per-thread state tracked in thread-local storage.
108struct Data {
109    /// Current collector state for the thread.
110    state: Cell<State>,
111    /// Last observed value of [EVENTS].
112    seen_events: Cell<usize>,
113}

115/// Releases the current thread from the collector when the thread exits.
116struct ExitGuard;

118impl Drop for ExitGuard {
119    fn drop(&mut self) {
120        release();
121    }
122}

149/// Call a closure with a reference to the current thread's collector state.
150///
151/// This hides the TLS access behind `hide` internally.
152#[inline(always)]
153fn data<R>(f: impl FnOnce(&Data) -> R) -> R {
154    DATA.with(|data| f(hide(data)))
155}

157/// Marks the current thread as pinned and returns a proof of that to the closure.
158///
159/// This adds the current thread to the set of threads that needs to regularly call [collect]
160/// before memory can be freed. [release] can be called if a thread no longer needs
161/// access to lock-free data structures for an extended period of time.
162///
163/// Nested calls to [pin] are allowed.
164///
165/// This will panic if called from a deferred callback.
166#[inline]
167pub fn pin<R>(f: impl FnOnce(Pin<'_>) -> R) -> R {
168    data(|data| {
169        if unlikely(!matches!(
170            data.state.get(),
171            State::Registered | State::Pinned
172        )) {
173            pin_cold();
174        }
175
176        let old_state = data.state.get();
177        data.state.set(State::Pinned);
178        let _guard = guard(old_state, |state| data.state.set(*state));
179        f(Pin {
180            _private: PhantomData,
181        })
182    })
183}
184
185#[inline(never)]
186#[cold]
187fn pin_cold() {
188    data(|data| match data.state.get() {
189        State::Unregistered => {
190            EXIT_GUARD.with(|_| ());
191            COLLECTOR.lock().register();
192            data.state.set(State::Registered);
193        }
194        State::Registered | State::Pinned => unreachable!(),
195        State::Collecting => cold_path(|| panic!("Deferred callbacks cannot call `pin`")),
196    })
197}

199/// Removes the current thread from the threads allowed to access lock-free data structures.
200///
201/// This allows memory to be freed without waiting for [collect] calls from the current thread.
202/// [pin] can be called after to continue accessing lock-free data structures.
203///
204/// This will not free any garbage so [collect] should be called before the last thread
205/// terminates to avoid memory leaks.
206///
207/// Calling this function when the thread is already unregistered is a no-op.
208///
209/// This will panic if called while the current thread is pinned or during a deferred callback.
210pub fn release() {
211    data(|data| match data.state.get() {
212        State::Unregistered => (),
213        State::Registered => {
214            data.state.set(State::Unregistered);
215            COLLECTOR.lock().unregister();
216        }
217        State::Pinned => cold_path(|| panic!("Cannot call `release` while pinned")),
218        State::Collecting => cold_path(|| panic!("Deferred callbacks cannot call `release`")),
219    })
220}

222/// Signals a quiescent state where garbage may be collected.
223///
224/// This may collect garbage using the callbacks registered in [Pin::defer_unchecked](struct.Pin.html#method.defer_unchecked).
225///
226/// This may panic if called while a thread is pinned or if called from a deferred callback.
227pub fn collect() {
228    data(|data| {
229        if cfg!(debug_assertions) {
230            // We check this only with `debug_assertions` to improve performance.
231            // A proper check is done in `collect_cold` if there's new events.
232            assert_collect_state(data);
233        }
234
235        // `EVENTS` can wrap around causing us to miss an event.
236        // That is unlikely and will just delay reclamation until another event is triggered.
237        let new = EVENTS.load(Ordering::Acquire);
238        if unlikely(new != data.seen_events.get()) {
239            collect_cold();
240        }
241    })
242}
243
244fn assert_collect_state(data: &Data) {
245    match data.state.get() {
246        State::Registered | State::Unregistered => (),
247        State::Pinned => panic!("Cannot call `collect` while pinned"),
248        State::Collecting => panic!("Deferred callbacks cannot call `collect`"),
249    }
250}
251
252#[inline(never)]
253#[cold]
254fn collect_cold() {
255    data(|data| {
256        assert_collect_state(data);
257
258        // Update seen events after `assert_collect_state` in it panics.
259        // This allows future `collect` to continue if we resume execution after the panic.
260        data.seen_events.set(EVENTS.load(Ordering::Relaxed));
261
262        let old_state = data.state.get();
263        let _guard = guard(old_state, |state| data.state.set(*state));
264        data.state.set(State::Collecting);
265
266        let callbacks = {
267            let mut collector = COLLECTOR.lock();
268
269            // Check if we could block any deferred methods
270            if let State::Registered = old_state {
271                collector.quiet()
272            } else {
273                collector.collect_unregistered()
274            }
275        };
276
277        let mut panic = None;
278
279        for callback in callbacks {
280            if let Err(payload) = panic::catch_unwind(AssertUnwindSafe(|| {
281                callback();
282            })) {
283                panic = Some(payload);
284            }
285        }
286
287        if let Some(payload) = panic {
288            panic::resume_unwind(payload)
289        }
290    })
291}

293static COLLECTOR: LazyLock<Mutex<Collector>> = LazyLock::new(|| Mutex::new(Collector::new()));

295type Callbacks = Vec<Box<dyn FnOnce() + Send>>;
296
297#[derive(Clone, Copy, PartialEq, Eq)]
298enum ThreadState {
299    /// The thread reported a quiescent state for the current epoch.
300    Quiet,
301    /// The thread still needs to report a quiescent state for the current epoch.
302    Busy,
303}
304
305/// Global collector state shared by all participating threads.
306struct Collector {
307    /// Callbacks that are ready to run on the next collection attempt.
308    pending: Callbacks,
309    /// Number of registered threads still marked as [ThreadState::Busy].
310    busy_count: usize,
311    /// Per-thread participation state for the current epoch.
312    threads: HashMap<ThreadId, ThreadState>,
313    /// Callbacks deferred during the current epoch.
314    current_deferred: Callbacks,
315    /// Callbacks that became eligible once the current epoch completes.
316    previous_deferred: Callbacks,
317}
318
319impl Collector {
320    fn new() -> Self {
321        Self {
322            pending: Vec::new(),
323            busy_count: 0,
324            threads: HashMap::new(),
325            current_deferred: Vec::new(),
326            previous_deferred: Vec::new(),
327        }
328    }
329
330    fn register(&mut self) {
331        self.busy_count += 1;
332        assert!(
333            self.threads
334                .insert(thread::current().id(), ThreadState::Busy)
335                .is_none()
336        );
337    }
338
339    fn unregister(&mut self) {
340        let state = self.threads.remove(&thread::current().id()).unwrap();
341        if state == ThreadState::Busy {
342            self.busy_count -= 1;
343
344            if self.busy_count == 0 {
345                self.complete_epoch(true);
346            }
347        } else if self.threads.is_empty() {
348            self.complete_epoch(true);
349        }
350    }
351
352    fn collect_unregistered(&mut self) -> Callbacks {
353        debug_assert!(!self.threads.contains_key(&thread::current().id()));
354
355        let mut callbacks = mem::take(&mut self.pending);
356
357        if self.threads.is_empty() {
358            callbacks.extend(mem::take(&mut self.previous_deferred));
359            callbacks.extend(mem::take(&mut self.current_deferred));
360        }
361
362        callbacks
363    }
364
365    fn quiet(&mut self) -> Callbacks {
366        let state = self.threads.get_mut(&thread::current().id()).unwrap();
367
368        let mut callbacks = mem::take(&mut self.pending);
369
370        if *state != ThreadState::Busy {
371            return callbacks;
372        }
373
374        self.busy_count -= 1;
375        *state = ThreadState::Quiet;
376
377        if self.busy_count == 0 {
378            // We pass `false` as we'll immediately take pending callbacks, so we don't need to signal them.
379            self.complete_epoch(false);
380            callbacks.extend(mem::take(&mut self.pending));
381
382            if !self.previous_deferred.is_empty() {
383                // We immediately call `quiet` so we will be up to date with the new epoch
384                DATA.with(|data| data.seen_events.set(EVENTS.load(Ordering::Relaxed)));
385
386                // Mark ourselves as quiet again
387                callbacks.extend(self.quiet());
388            }
389        }
390
391        callbacks
392    }
393
394    fn defer(&mut self, callback: Box<dyn FnOnce() + Send>) {
395        self.current_deferred.push(callback);
396        EVENTS.fetch_add(1, Ordering::Release);
397    }
398
399    fn complete_epoch(&mut self, signal_pending: bool) {
400        self.pending.extend(mem::take(&mut self.previous_deferred));
401
402        if self.threads.is_empty() {
403            self.pending.extend(mem::take(&mut self.current_deferred));
404            if !self.pending.is_empty() {
405                // Signal future threads to check in
406                EVENTS.fetch_add(1, Ordering::Release);
407            }
408            return;
409        }
410
411        self.busy_count = self.threads.len();
412        self.threads.values_mut().for_each(|value| {
413            *value = ThreadState::Busy;
414        });
415        self.previous_deferred = mem::take(&mut self.current_deferred);
416
417        if (signal_pending && !self.pending.is_empty()) || !self.previous_deferred.is_empty() {
418            // Signal all threads to check in
419            EVENTS.fetch_add(1, Ordering::Release);
420        }
421    }
422}
